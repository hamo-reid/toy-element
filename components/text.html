Text 文本 ​

常见文本操作

基础用法 ​

<template>
  <p>
    <hm-text>Default</hm-text>
    <hm-text type="primary" style="margin: 0 0.25rem;">Primary</hm-text>
    <hm-text type="success" style="margin: 0 0.25rem;">Success</hm-text>
    <hm-text type="warning" style="margin: 0 0.25rem;">Warning</hm-text>
    <hm-text type="danger" style="margin: 0 0.25rem;">Danger</hm-text>
    <hm-text type="info" style="margin: 0 0.25rem;">Info</hm-text>
  </p>
</template>

尺寸 ​

<template>
  <hm-text size="large" style="margin: 0 0.25rem;">Large</hm-text>
  <hm-text size="default" style="margin: 0 0.25rem;">Default</hm-text>
  <hm-text size="small" style="margin: 0 0.25rem;">subscript</hm-text>
</template>

省略 ​

<template>
  <div style="width: 150px;">
    <hm-text :truncated="true">这是一段很长的文本，需要被截断</hm-text>
  </div>
</template>

自定义标签 ​

<template>
  <div style="display: flex;flex-direction: column; align-items: center">
    <hm-text>span</hm-text>
    <hm-text tag="p">This is a paragraph.</hm-text>
    <hm-text tag="b">Bold</hm-text>
    <hm-text tag="i">Italic</hm-text>
    <hm-text>
      This is
      <hm-text tag="sub" size="small">subscript</hm-text>
    </hm-text>
    <hm-text>
      This is
      <hm-text tag="sup" size="small">superscript</hm-text>
    </hm-text>
    <hm-text tag="ins">Inserted</hm-text>
    <hm-text tag="del">Deleted</hm-text>
    <hm-text tag="mark">Marked</hm-text>
  </div>
</template>
